private IEnumerator FadeInRoutine()
{
    // 遷移を感じさせるため、一瞬（0.1秒）黒い状態を維持
    yield return new WaitForSecondsRealtime(0.1f);

    float duration = 1.0f; // 1秒かけてフェード
    float elapsed = 0f;

    while (elapsed < duration)
    {
        elapsed += Time.unscaledDeltaTime;
        float t = elapsed / duration;
        
        // イーズアウト（Ease-Out）処理: 1 - (1-t)^2
        // 最初は速く、後半はゆっくり透明になる
        float easeT = 1f - Mathf.Pow(1f - t, 2f);
        
        // 不透明度（opacity）を 1.0 から 0.0 へ減少させる
        _fadeOverlay.style.opacity = 1f - easeT;
        yield return null;
    }

    // 完全に透明になったら要素を非表示にする
    _fadeOverlay.style.opacity = 0f;
    _fadeOverlay.style.display = DisplayStyle.None;
}

구현 포인트

  • 이징（Easing）: 단순한 선형 보간이 아닌, 1f - Mathf.Pow(1f - t, 2f) 라는 계산식을 사용함으로써, 보다 부드럽고 자연스러운 페이드인을 실현하고 있습니다.
  • Unscaled Time: Time.unscaledDeltaTime 을 사용하고 있기 때문에, 게임 정지（Time.timeScale = 0）등의 영향을 받지 않고 애니메이션이 재생됩니다.
  • UITKの활용: C#에서 style.opacity 나 style.display 를 직접 조작함으로써, 셰이더 등을 작성하지 않고 UI 시스템만으로 완결시키고 있습니다.
